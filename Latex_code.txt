\begin{frame}{Econometric model}
 \begin{itemize}
\item{So, as we can see the impact of supply and demand shock on equilibrium depends on the elasticises, as we can show}:
\end{itemize}

\begin{equation}
    \Delta h_{t}^{l} = (\frac{1}{1-(\frac{\alpha^{l}}{\beta^{l}})^{-1}})\epsilon_{d,t}^{l}+ (\frac{1}{1-\frac{\alpha^{l}}{\beta^{l}}})\epsilon_{s,t}^{l}
\end{equation}

\begin{equation}
    \Delta w_{t}^{l} = (\frac{\frac{1}{\beta^{l}}}{\frac{\alpha^{l}{\beta^{l}}}-1}-1)\epsilon_{d,t}^{l}+(\frac{\frac{1}{\beta^{l}}}{1-\frac{\alpha^{l}}{\beta^{l}}})\epsilon_{s,t}^{l}
\end{equation}

\begin{itemize}
\item{Hence, if $\beta^{l} < 0$ and $\alpha^{l} > 0$:}
\end{itemize}

\begin{center}
$\frac{\delta \Delta h_{t}^{l}}{\epsilon_{d,t}^{l}} > 0$ and $\frac{\delta \Delta w_{t}^{l}}{\epsilon_{d,t}^{l}} > 0$
\end{center}
\begin{center}
$\frac{\delta \Delta h_{t}^{l}}{\epsilon_{s,t}^{l}} > 0$ and $\frac{\delta \Delta w_{t}^{l}}{\epsilon_{s,t}^{l}} < 0$  
\end{center}

\end{frame}

\begin{frame}{Bayesian specification}

\begin{itemize}
\item{The reduced-form vector autoregression (VAR) associated with the SVAR model 2:}
\end{itemize}

\begin{equation}
 y_{t}^{l} = \Pi_{0}^{l}+\Pi_{0}^{l}(L) y_{t-1}^{l}+\nu_{t}^{l}
\end{equation}

\begin{equation*}
\Pi_{0}^{l} = (A^{l})^{-1}B_{0}^{l} 
\end{equation*}

\begin{equation*}
\Pi^{l}(L) = (A^{l})^{-1}B^{l}(L) 
\end{equation*}

\begin{equation*}
\nu_{t}^{l} = (A^{l})^{-1}\epsilon_{0}^{l} 
\end{equation*}

\begin{equation*}
E[\nu_{t}^{l}(\nu_{t}^{l})^{'}]=(A^{l})^{-1}D((A^{l})^{-1})^{'}
\end{equation*}

\begin{itemize}
    \item So, prior beliefs about the parameters are represented by a joint density $ p(A,D,B). $
\end{itemize}


\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{Bayesian specification}
\begin{itemize}
    \item {Baumeister and Hamilton (2015) developed an algorithm that estimates the parameters
of an SVAR model using Bayesian methods (SBVAR). Their algorithm applies a
random-walk Metropolis-Hastings algorithm to seek elasticity values for A, considering prior
information, that diagonalizes the covariance matrix of the reduced form errors.}
\end{itemize}
\begin{itemize}
    \item {Baumeister and Hamilton (2015) prove that the posterior can be written as:}
\end{itemize}

\begin{equation}
p(A,D,B | Y_{T}) = p(A|Y_{T})p(D|A,Y_{T})p(B|A,D,Y_{T})
\end{equation}

\begin{itemize}
    \item {Where $ p(B) = A\Pi$ is the coefficients set from Bayesian VAR Minnesota priors of the reduced form.}
\end{itemize}

\begin{itemize}
    \item {Where the elements of  $ D $  and the conditional posterior distribution  $p(D| A, Y_{t} )$ are gamma distributed.}
\end{itemize}

\begin{itemize}
    \item {Finally, $p(A|Y_{T})$ is estimated by a random-walk Metropolis-Hastings
algorithm.}
\end{itemize}
\end{frame}

\begin{frame}{Methodology intuition}

 \begin{itemize}

\item {If it is observed that employment and wages (prices and quantities) move in the same direction, it is more likely that these movements are caused by a demand shock.
On the other hand, if we observe that employment and wages move in opposite directions, it is assigned to a supply shock.}

\item {Uhlig (2017) explain the basic required assumptions for identifying demand and supply shocks. Actually, It may be other shocks that impact to demand and supply. Without loss of generality, the approach of Baumeister y Hamilton (2015), and it is used by Brinca, Duarte and Faria e Castro (2020), those other shocks do not affect demand and supply in a systematic way.}

 \end{itemize}
\end{frame}
